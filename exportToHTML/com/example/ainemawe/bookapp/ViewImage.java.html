44       @Override 
45       public void onCreate(Bundle savedInstanceState) { 
46           super.onCreate(savedInstanceState); 
47           // Get the view from view_image.xml 
48           setContentView(R.layout.view_image); 
49    
50           //buttons 
51           emailBtn = (Button) findViewById(R.id.email_btn); 
52           emailBtn.setOnClickListener(ViewImage.this); 
53           deleteBtn = (Button) findViewById(R.id.trash_btn); 
54           deleteBtn.setOnClickListener(ViewImage.this); 
55           amazonBtn = (Button) findViewById(R.id.amazon_btn); 
56           amazonBtn.setOnClickListener(this); 
57           academicLibraryBtn = (Button) findViewById(R.id.academic_library_btn); 
58           academicLibraryBtn.setOnClickListener(this); 
59           publicLibraryBtn = (Button) findViewById(R.id.public_library_btn); 
60           publicLibraryBtn.setOnClickListener(this); 
61    
62    
63           // Retrieve data from ViewSaved on ListView item click 
64           Intent i = getIntent(); 
65           // Get the position 
66           position = i.getExtras().getInt("position"); 
67           // Get String arrays FilePathStrings 
68           filepath = i.getStringArrayExtra("filepath"); 
69           // Get String arrays FileNameStrings 
70           String[] filename = i.getStringArrayExtra("filename"); 
71           title = filename[position].substring(0, filename[position].length() - 4); 
72    
73    
74           // Locate the ImageView in view_image.xml 
75           imageview = (ImageView) findViewById(R.id.full_image_view); 
76           // Decode the filepath with BitmapFactory followed by the position 
77           Bitmap bmp = BitmapFactory.decodeFile(filepath[position]); 
78           // Set the decoded bitmap into ImageView 
79           imageview.setImageBitmap(bmp); 
80    
81       }

83       @Override 
84       public boolean onCreateOptionsMenu(Menu menu) {

97       @Override 
98       public void onClick(View v) { 
99           if (v.getId() == R.id.trash_btn) { 
100  //check for confirmation and delete 
101              new AlertDialog.Builder(ViewImage.this) 
102                      .setTitle("Delete entry") 
103                      .setMessage("Are you sure you want to delete this entry?") 
104                      .setPositiveButton(android.R.string.yes, new DialogInterface.OnClickListener() { 
105                          public void onClick(DialogInterface dialog, int which) { 
106                              File image = new File(filepath[position]); 
107                              image.delete(); 
108                              Toast.makeText(getApplicationContext(), "Deleted!", Toast.LENGTH_SHORT).show(); 
109                          } 
110                      }) 
111                      .setNegativeButton(android.R.string.no, new DialogInterface.OnClickListener() { 
112                          public void onClick(DialogInterface dialog, int which) { 
113                              // do nothing 
114                          } 
115                      }) 
116                      .setIcon(android.R.drawable.ic_dialog_alert) 
117                      .show(); 
118   
119   
120          } else if (v.getId() == R.id.email_btn) { 
121              //set email settings 
122              Intent emailIntent = new Intent(android.content.Intent.ACTION_SEND); 
123              emailIntent.putExtra(android.content.Intent.EXTRA_TEXT, "I think you may find this book interesting"); 
124              emailIntent.putExtra(android.content.Intent.EXTRA_SUBJECT, "Book Recommendation"); 
125              emailIntent.setData(Uri.parse("mailto:" + " ")); 
126              emailIntent.setType("text/html"); 
127              File image = new File("file://" + filepath[position]); 
128              Uri uri = Uri.parse("file://" + image); 
129              emailIntent.putExtra(Intent.EXTRA_STREAM, uri); 
130              startActivity(Intent.createChooser(emailIntent, "Choose an Email client"));

132          }

164      private void setShareIntent() { 
165   
166          if (myShareActionProvider != null) { 
167              // create an Intent with the contents 
168              Intent shareIntent = new Intent(Intent.ACTION_SEND); 
169              File image = new File("file://" + filepath[position]); 
170              Uri uri = Uri.parse("file://" + image); 
171              shareIntent.setType("text/html"); 
172              shareIntent.putExtra(Intent.EXTRA_STREAM, uri); 
173              shareIntent.putExtra(Intent.EXTRA_SUBJECT, "Book Recommendation"); 
174              myShareActionProvider.setShareIntent(shareIntent); 
175          } 
176      }
