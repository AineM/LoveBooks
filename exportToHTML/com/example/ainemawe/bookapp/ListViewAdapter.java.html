24   public class ListViewAdapter extends BaseAdapter { 
25       private static LayoutInflater inflater = null; 
26       // Declare variables 
27       private Activity activity; 
28       private String[] filepath; 
29       private String[] filename; 
30    
31       public ListViewAdapter(Activity a, String[] fpath, String[] fname) { 
32           activity = a; 
33           filepath = fpath; 
34           filename = fname; 
35           inflater = (LayoutInflater) activity 
36                   .getSystemService(Context.LAYOUT_INFLATER_SERVICE); 
37    
38       } 
39    
40       public int getCount() { 
41           return filepath.length; 
42    
43       } 
44    
45       public Object getItem(int position) { 
46           return position; 
47       } 
48    
49       public long getItemId(int position) { 
50           return position; 
51       } 
52    
53       public View getView(int position, View convertView, ViewGroup parent) { 
54           View vi = convertView; 
55           if (convertView == null) 
56               vi = inflater.inflate(R.layout.listview_item, null); 
57           // Locate the TextView in listview_item.xml 
58           TextView text = (TextView) vi.findViewById(R.id.Itemname); 
59           // Locate the ImageView in listview_item.xml 
60           ImageView image = (ImageView) vi.findViewById(R.id.image); 
61    
62           // get file name and use substring to extract title and set to the TextView 
63           String bookTitle = filename[position]; 
64           text.setText(bookTitle.substring(0, bookTitle.length() - 4)); 
65    
66   //TODO Fix occasional infrequent overloads and gives java.lang.OutOfMemoryError - compress image 
67    
68           // Decode the filepath with BitmapFactory followed by the position 
69           Bitmap bmp = BitmapFactory.decodeFile(filepath[position]); 
70    
71           // Set the decoded bitmap into ImageView 
72           image.setImageBitmap(bmp); 
73    
74    
75           return vi; 
76    
77    
78       } 
79   }
