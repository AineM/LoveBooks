184      @Override 
185      public void onActivityResult(int requestCode, int resultCode, Intent intent) { 
186          IntentResult result = IntentIntegrator.parseActivityResult(requestCode, resultCode, intent); 
187   
188          //check result 
189          if (result != null) { 
190              scanContent = result.getContents(); 
191              String scanFormat = result.getFormatName(); 
192             // result.getOrientation(); 
193              if (scanContent != null && scanFormat != null && scanFormat.equalsIgnoreCase("EAN_13")) { 
194                  String bookSearchString = GOOGLE_APIS_BOOKS_URL + scanContent + GOOGLE_APIS_KEY; 
195                  new GetBookDetails().execute(bookSearchString); 
196   
197   
198              } else { 
199                  Toast toast = Toast.makeText(getApplicationContext(), 
200                          "Not a valid scan!", Toast.LENGTH_SHORT); 
201                  toast.show(); 
202                  startActivity(new Intent(getApplicationContext(), Home.class)); 
203   
204   
205              } 
206              //Log.v("SCAN", "content: " + scanContent + " - format: " + scanFormat); 
207          } else { 
208              Toast toast = Toast.makeText(getApplicationContext(), 
209                      "No scan data received!", Toast.LENGTH_SHORT); 
210              toast.show(); 
211              startActivity(new Intent(getApplicationContext(), Home.class)); 
212   
213          }

343      private class GetBookDetails extends AsyncTask<String, Void, String> { 
344   
345   
346          /** 
347           * 
348            * @param bookURLs 
349           * @return 
350           */ 
351          @Override 
352          protected String doInBackground(String... bookURLs) { 
353              StringBuilder bookBuilder = new StringBuilder(); 
354              for (String bookSearchURL : bookURLs) { 
355                  //search urls 
356                  HttpClient bookClient = new DefaultHttpClient(); 
357                  try { 
358                      //get the data 
359                      HttpGet bookGet = new HttpGet(bookSearchURL); 
360                      HttpResponse bookResponse = bookClient.execute(bookGet); 
361                      StatusLine bookSearchStatus = bookResponse.getStatusLine(); 
362                      if (bookSearchStatus.getStatusCode() == 200) { 
363                          //we have a result 
364                          HttpEntity bookEntity = bookResponse.getEntity(); 
365                          InputStream bookContent = bookEntity.getContent(); 
366                          InputStreamReader bookInput = new InputStreamReader(bookContent); 
367                          BufferedReader bookReader = new BufferedReader(bookInput); 
368   
369                          String lineIn; 
370                          while ((lineIn = bookReader.readLine()) != null) { 
371                              bookBuilder.append(lineIn); 
372                          } 
373                      } 
374                  } catch (Exception e) { 
375                      e.printStackTrace(); 
376                  } 
377              } 
378              //return results 
379              return bookBuilder.toString(); 
380          } 
381   
382          protected void onPostExecute(String result) { 
383              //parse search results 
384              try { 
385                  JSONObject resultObject = new JSONObject(result); 
386                  JSONArray bookArray = resultObject.getJSONArray("items"); 
387                  //get first result 
388                  JSONObject bookObject = bookArray.getJSONObject(0); 
389                  JSONObject volumeObject = bookObject.getJSONObject("volumeInfo"); 
390   
391   
392                  //attempt to download title found 
393   
394                  try { 
395                      title = volumeObject.getString("title"); 
396                      titleText.setText("TITLE: " + title); 
397   
398                  } catch (JSONException jse) { 
399                      titleText.setText(""); 
400                      jse.printStackTrace(); 
401                  } 
402   
403                  //attempt to download author found 
404                  StringBuilder authorBuild = new StringBuilder(""); //for more than one author 
405                  try { 
406                      JSONArray authorArray = volumeObject.getJSONArray("authors"); 
407                      for (int a = 0; a < authorArray.length(); a++) { 
408                          if (a > 0) authorBuild.append(", "); 
409                          authorBuild.append(authorArray.getString(a)); 
410                      } 
411                      author = authorBuild.toString(); 
412                      authorText.setText("AUTHOR(S): " + author); 
413                  } catch (JSONException jse) { 
414                      authorText.setText(""); 
415                      jse.printStackTrace(); 
416                  } 
417   
418   
419                  //attempt to download publisher 
420                  try { 
421                      publisherText.setText("PUBLISHER: " + volumeObject.getString("publisher")); 
422                  } catch (JSONException jse) { 
423                      publisherText.setText("PUBLISHER: ?"); 
424                      jse.printStackTrace(); 
425                  } 
426   
427   
428                  //attempt to download publication date 
429                  try { 
430                      dateText.setText("DATE PUBLISHED: " + volumeObject.getString("publishedDate")); 
431                  } catch (JSONException jse) { 
432                      dateText.setText(""); 
433                      jse.printStackTrace(); 
434                  } 
435   
436                  //attempt to download page number 
437                  try { 
438                      pageText.setText("PAGES: " + volumeObject.getString("pageCount")); 
439                  } catch (JSONException jse) { 
440                      pageText.setText(""); 
441                      jse.printStackTrace(); 
442                  } 
443   
444                  //attempt to download categories found 
445                  StringBuilder categoriesBuild = new StringBuilder(""); //for more than one author 
446                  try { 
447                      JSONArray categoriesArray = volumeObject.getJSONArray("categories"); 
448                      for (int a = 0; a < categoriesArray.length(); a++) { 
449                          if (a > 0) categoriesBuild.append(", "); 
450                          categoriesBuild.append(categoriesArray.getString(a)); 
451                      } 
452                      categoriesText.setText("CATEGORIES: " + categoriesBuild.toString()); 
453                  } catch (JSONException jse) { 
454                      categoriesText.setText(""); 
455                      jse.printStackTrace(); 
456                  } 
457   
458   
459                  //attempt to download description 
460                  try { 
461                      descriptionText.setText("DESCRIPTION: " + volumeObject.getString("description")); 
462                  } catch (JSONException jse) { 
463                      descriptionText.setText(""); 
464                      jse.printStackTrace(); 
465                  } 
466   
467                  //star rating 
468                  try { 
469                      //set stars 
470                      double decNumStars = Double.parseDouble(volumeObject.getString("averageRating")); 
471                      int numStars = (int) decNumStars; 
472                      starLayout.setTag(numStars); 
473                      starLayout.removeAllViews(); 
474                      for (int s = 0; s < numStars; s++) { 
475                          starViews[s].setImageResource(R.drawable.star); 
476                          starLayout.addView(starViews[s]); 
477                      } 
478                  } catch (JSONException jse) { 
479                      starLayout.removeAllViews(); 
480                      jse.printStackTrace(); 
481                  } 
482                  try { 
483                      ratingCountText.setText(" - " + volumeObject.getString("ratingsCount") + " ratings"); 
484                  } catch (JSONException jse) { 
485                      ratingCountText.setText(""); 
486                      jse.printStackTrace(); 
487                  } 
488   
489                  //link button 
490                  try { 
491                      linkBtn.setTag(volumeObject.getString("infoLink")); 
492                      linkBtn.setVisibility(View.VISIBLE); 
493                  } catch (JSONException jse) { 
494                      linkBtn.setVisibility(View.GONE); 
495                      jse.printStackTrace(); 
496                  } 
497   
498                  saveBtn.setVisibility(View.VISIBLE); 
499                  amazonBtn.setVisibility(View.VISIBLE); 
500                  academicLibraryBtn.setVisibility(View.VISIBLE); 
501                  saveBtn.setVisibility(View.VISIBLE); 
502                  publicLibraryBtn.setVisibility(View.VISIBLE); 
503                  isbnText.setText("ISBN: " + scanContent); 
504   
505   
506                  //attempt to download image or set placeholder 
507                  try { 
508                      JSONObject imageInfo = volumeObject.getJSONObject("imageLinks"); 
509                      new GetCover().execute(imageInfo.getString("smallThumbnail")); 
510                  } catch (JSONException jse) { 
511                      //thumbView.setImageBitmap(null); 
512                      Drawable myDrawable = getResources().getDrawable(R.drawable.book_cover); 
513                      thumbView.setImageDrawable(myDrawable); 
514                      jse.printStackTrace(); 
515                  } 
516   
517              } catch (Exception e) { 
518                  //what to display if no result retrieved 
519                  e.printStackTrace(); 
520                  titleText.setText("NOT FOUND"); 
521                  authorText.setText(""); 
522                  publisherText.setText(""); 
523                  descriptionText.setText(""); 
524                  pageText.setText(""); 
525                  dateText.setText(""); 
526                  starLayout.removeAllViews(); 
527                  ratingCountText.setText(""); 
528                  thumbView.setImageBitmap(null); 
529                  //BitmapFactory.decodeFile 
530   
531                  linkBtn.setVisibility(View.GONE); 
532              } 
533          }

535          /** 
536           *get cover image 
537           */ 
538           private class GetCover extends AsyncTask<String, Void, String> { 
539              @Override 
540              protected String doInBackground(String... thumbURLs) { 
541                  //attempt to download image 
542                  try { 
543                      URL thumbURL = new URL(thumbURLs[0]); 
544                      URLConnection thumbConn = thumbURL.openConnection(); 
545                      thumbConn.connect(); 
546                      InputStream thumbIn = thumbConn.getInputStream(); 
547                      BufferedInputStream thumbBuff = new BufferedInputStream(thumbIn); 
548                      thumbImg = BitmapFactory.decodeStream(thumbBuff); 
549                      thumbBuff.close(); 
550                      thumbIn.close(); 
551                  } catch (Exception e) { 
552                      e.printStackTrace(); 
553                  } 
554                  return ""; 
555   
556              } 
557   
558              protected void onPostExecute(String result) { 
559                  thumbView.setImageBitmap(thumbImg); 
560              } 
561   
562          }
