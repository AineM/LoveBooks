300      private class GetBookDetails extends AsyncTask<String, Void, String> { 
301   
302          /** retrieve  book info  */ 
303          @Override 
304          protected String doInBackground(String... bookURLs) { 
305              StringBuilder bookBuilder = new StringBuilder(); 
306              for (String bookSearchURL : bookURLs) { 
307                  //search urls 
308                  HttpClient bookClient = new DefaultHttpClient(); 
309                  try { 
310                      //get the data 
311                      HttpGet bookGet = new HttpGet(bookSearchURL); 
312                      HttpResponse bookResponse = bookClient.execute(bookGet); 
313                      StatusLine bookSearchStatus = bookResponse.getStatusLine(); 
314                      if (bookSearchStatus.getStatusCode() == 200) { 
315                          //we have a result 
316                          HttpEntity bookEntity = bookResponse.getEntity(); 
317                          InputStream bookContent = bookEntity.getContent(); 
318                          InputStreamReader bookInput = new InputStreamReader(bookContent); 
319                          BufferedReader bookReader = new BufferedReader(bookInput); 
320   
321                          String lineIn; 
322                          while ((lineIn = bookReader.readLine()) != null) { 
323                              bookBuilder.append(lineIn); 
324                          } 
325                      } 
326                  } catch (Exception e) { 
327                      e.printStackTrace(); 
328                  } 
329              } 
330              //return results 
331              return bookBuilder.toString(); 
332          } 
333   
334          protected void onPostExecute(String result) { 
335              //parse search results 
336              try { 
337                  JSONObject resultObject = new JSONObject(result); 
338                  JSONArray bookArray = resultObject.getJSONArray("items"); 
339                  //get first result 
340                  JSONObject bookObject = bookArray.getJSONObject(0); 
341                  JSONObject volumeObject = bookObject.getJSONObject("volumeInfo"); 
342   
343                  //attempt to download title found 
344   
345                  try { 
346                      title = volumeObject.getString("title"); 
347                      titleText.setText("TITLE: " + title); 
348   
349                  } catch (JSONException jse) { 
350                      titleText.setText(""); 
351                      jse.printStackTrace(); 
352                  } 
353   
354                  //attempt to download author found 
355                  StringBuilder authorBuild = new StringBuilder(""); //for more than one author 
356                  try { 
357                      JSONArray authorArray = volumeObject.getJSONArray("authors"); 
358                      for (int a = 0; a < authorArray.length(); a++) { 
359                          if (a > 0) authorBuild.append(", "); 
360                          authorBuild.append(authorArray.getString(a)); 
361                      } 
362                      author = authorBuild.toString(); 
363                      authorText.setText("AUTHOR(S): " + author); 
364                  } catch (JSONException jse) { 
365                      authorText.setText(""); 
366                      jse.printStackTrace(); 
367                  } 
368   
369   
370                  //attempt to download publisher 
371                  try { 
372                      publisherText.setText("PUBLISHER: " + volumeObject.getString("publisher")); 
373                  } catch (JSONException jse) { 
374                      publisherText.setText("PUBLISHER: ?"); 
375                      jse.printStackTrace(); 
376                  } 
377   
378   
379                  //attempt to download publication date 
380                  try { 
381                      String publicationDate = volumeObject.getString("publishedDate"); 
382                      dateText.setText("DATE PUBLISHED: " + publicationDate); 
383                  } catch (JSONException jse) { 
384                      dateText.setText(""); 
385                      jse.printStackTrace(); 
386                  } 
387   
388                  //attempt to download page number 
389                  try { 
390                      String pageCount = volumeObject.getString("pageCount"); 
391                      pageText.setText("PAGES: " + pageCount); 
392                  } catch (JSONException jse) { 
393                      pageText.setText(""); 
394                      jse.printStackTrace(); 
395                  } 
396   
397                  //attempt to download categories found 
398                  StringBuilder categoriesBuild = new StringBuilder(""); //for more than one author 
399                  try { 
400                      JSONArray categoriesArray = volumeObject.getJSONArray("categories"); 
401                      for (int a = 0; a < categoriesArray.length(); a++) { 
402                          if (a > 0) categoriesBuild.append(", "); 
403                          categoriesBuild.append(categoriesArray.getString(a)); 
404                      } 
405                      categoriesText.setText("CATEGORIES: " + categoriesBuild.toString()); 
406                  } catch (JSONException jse) { 
407                      categoriesText.setText(""); 
408                      jse.printStackTrace(); 
409                  } 
410   
411                  //attempt to download description 
412                  try { 
413                      descriptionText.setText("DESCRIPTION: " + volumeObject.getString("description")); 
414                  } catch (JSONException jse) { 
415                      descriptionText.setText(""); 
416                      jse.printStackTrace(); 
417                  } 
418   
419                  //star rating 
420                  try { 
421                      //set stars 
422                      double decNumStars = Double.parseDouble(volumeObject.getString("averageRating")); 
423                      int numStars = (int) decNumStars; 
424                      starLayout.setTag(numStars); 
425                      starLayout.removeAllViews(); 
426                      for (int s = 0; s < numStars; s++) { 
427                          starViews[s].setImageResource(R.drawable.star); 
428                          starLayout.addView(starViews[s]); 
429                      } 
430                  } catch (JSONException jse) { 
431                      starLayout.removeAllViews(); 
432                      jse.printStackTrace(); 
433                  } 
434                  try { 
435                      ratingCountText.setText(" - " + volumeObject.getString("ratingsCount") + " ratings"); 
436                  } catch (JSONException jse) { 
437                      ratingCountText.setText(""); 
438                      jse.printStackTrace(); 
439                  } 
440   
441                  //link button 
442                  try { 
443                      linkBtn.setTag(volumeObject.getString("infoLink")); 
444                      linkBtn.setVisibility(View.VISIBLE); 
445                  } catch (JSONException jse) { 
446                      linkBtn.setVisibility(View.GONE); 
447                      jse.printStackTrace(); 
448                  } 
449   
450                  saveBtn.setVisibility(View.VISIBLE); 
451                  amazonBtn.setVisibility(View.VISIBLE); 
452                  academicLibraryBtn.setVisibility(View.VISIBLE); 
453                  saveBtn.setVisibility(View.VISIBLE); 
454                  publicLibraryBtn.setVisibility(View.VISIBLE); 
455                  isbnText.setText(queryISBN); 
456   
457   
458                  //attempt to download image 
459                  try { 
460                      JSONObject imageInfo = volumeObject.getJSONObject("imageLinks"); 
461                      new GetCover().execute(imageInfo.getString("smallThumbnail")); 
462                  } catch (JSONException jse) { 
463                      Drawable myDrawable = getResources().getDrawable(R.drawable.book_cover); 
464                      thumbView.setImageDrawable(myDrawable); 
465                      //thumbView.setImageBitmap(null); 
466                      jse.printStackTrace(); 
467                  } 
468   
469              } catch (Exception e) { 
470                  //what to display if no result retrieved 
471                  e.printStackTrace(); 
472                  titleText.setText("NOT FOUND"); 
473                  authorText.setText(""); 
474                  publisherText.setText(""); 
475                  descriptionText.setText(""); 
476                  pageText.setText(""); 
477                  dateText.setText(""); 
478                  starLayout.removeAllViews(); 
479                  ratingCountText.setText(""); 
480                  thumbView.setImageBitmap(null); 
481                  linkBtn.setVisibility(View.GONE); 
482              } 
483          }

485          //get cover image 
486          private class GetCover extends AsyncTask<String, Void, String> { 
487              @Override 
488              protected String doInBackground(String... thumbURLs) { 
489                  //attempt to download image 
490                  try { 
491                      URL thumbURL = new URL(thumbURLs[0]); 
492                      URLConnection thumbConn = thumbURL.openConnection(); 
493                      thumbConn.connect(); 
494                      InputStream thumbIn = thumbConn.getInputStream(); 
495                      BufferedInputStream thumbBuff = new BufferedInputStream(thumbIn); 
496                      thumbImg = BitmapFactory.decodeStream(thumbBuff); 
497                      thumbBuff.close(); 
498                      thumbIn.close(); 
499                  } catch (Exception e) { 
500                      e.printStackTrace(); 
501                  } 
502                  return ""; 
503   
504              } 
505   
506              protected void onPostExecute(String result) { 
507                  thumbView.setImageBitmap(thumbImg); 
508              } 
509   
510          }
