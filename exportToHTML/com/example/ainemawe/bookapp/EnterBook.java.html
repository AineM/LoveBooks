26           super.onCreate(savedInstanceState); 
27           setContentView(R.layout.activity_enter_book);

29           //text fields

34           //button 
35           buttonSearch = (Button) findViewById(R.id.buttonSearch);

42       @Override 
43       public void onClick(View v) { 
44           if (v.getId() == R.id.buttonSearch) { 
45               //Use inputted text for search 
46               String queryISBN = editISBN.getText().toString(); 
47               if (queryISBN.length() == 13 || queryISBN.length() == 10) { 
48                   Intent intent = new Intent(EnterBook.this, Search.class); 
49                   intent.putExtra(EXTRA_MESSAGE, queryISBN); 
50                   startActivity(intent); 
51               } else { 
52                   new AlertDialog.Builder(EnterBook.this) 
53                           .setTitle("ISBN Not Valid") 
54                           .setMessage("ISBN must be 10 or 13 digits") 
55                           .setPositiveButton(android.R.string.yes, 
56                                   new DialogInterface.OnClickListener() { 
57                                       public void onClick(DialogInterface dialog, int which) { 
58                                           dialog.dismiss(); 
59                                       } 
60                                   }) 
61                           .setIcon(android.R.drawable.ic_dialog_alert) 
62                           .show(); 
63    
64                   //Toast.makeText(this, "ISBN must be 10 or 13 digits", Toast.LENGTH_LONG).show(); 
65               } 
66           } else if (v.getId() == R.id.buttonCancel){ 
67               startActivity(new Intent(getApplicationContext(), Home.class)); 
68           } 
69       }
